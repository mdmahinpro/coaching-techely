export async function sendSMS(phones, message, type='custom') {
  const phones_arr = Array.isArray(phones) ? phones : [phones];
  const { data: settings } = await supabaseAdmin
    .from('site_settings')
    .select('key,value')
    .in('key',['mimsms_api_key','mimsms_sender_id','institute_name']);

  const cfg = Object.fromEntries(settings.map(s=>[s.key,s.value]));

  const res = await fetch('https://api.mimsms.com/smsapi',{
    method:'POST',
    headers:{'Content-Type':'application/json'},
    body: JSON.stringify({
      api_key: cfg.mimsms_api_key,
      senderid: cfg.mimsms_sender_id || 'CoachMS',
      type: 'text',
      scheduleat: '',
      route: 'T',
      contacts: phones_arr.join(','),
      msg: message
    })
  });
  const data = await res.json();

  // Log to DB
  await supabaseAdmin.from('sms_logs').insert(
    phones_arr.map(p=>({ phone:p, message, type, status: data.status==='success'?'sent':'failed', response:data }))
  );
  return data;
}

export function fillTemplate(template, vars) {
  return template.replace(/\{([^}]+)\}/g, (_,k) => vars[k] || '');
}
